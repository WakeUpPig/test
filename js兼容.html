6）window.getComputedStyle(元素,元素的伪类(没有写null))
   currentStyle
    function getCss(ele,attr){
        if(window.getComputedStyle){
            return window.getComputedStyle(ele,null)[attr];
            }else{
                return ele.currentStyle[attr];
            }
    }
7）scrollWidth/scrollHeight:如果内容没有溢出的话，相当于client值是一样的（上
   padding+实际内容的高度）
   如果内容溢出的话是只加上padding，下padding不加了
8）e.pageX e.pageY
   e.relatedTarget
   var target = e.target|| e.srcElement;
   ele.attachEvent   ele.addEventListener(type,fn,false);
   ele.detachEvent   ele.removeEventListener(type,fn,false);
   e.preventDefault  e.returnValue=false;
   e.stopPropagation? e.stopPropagation(): e.cancelBubble=true;
   e=e||window.event;
   var type = e.type;
   e.pageX=(document.documentElement.scrollLeft||document.body.scrollLeft)+ e.clientX;
   e.pageY=(document.documentElement.scrollTop||document.body.scrollTop)+ e.clientY;
   e.stopPropagation=function(){
      e.cancelable=true;
   };
   e.preventDefault=function(){
      e.returnValue=false;
   };
